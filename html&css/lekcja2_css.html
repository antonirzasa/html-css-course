Podstawowe właściwości CSS

  • text-align: (wyrównanie tekstu)
    • left; (do lewej)
    • center; (do środka)
    • right; (do prawej)
  • text-decoration: (dekorowanie tekstu)
    • none; (brak)
    • underline; (podkreślenie)
    • overine; (nadkreślenie)
    • line-through; (przekreślenie)
  • font-family: (rodzaj czcionki)
    • NAZWA CZCIONKI;
  • font-size: (rozmiar czcionki)
    • (ilość)px;
    • (ilość)%;
  • font-weight: (grubość czcionki)
    • normal; (normalna)
    • bold; (pogrubiona)
    • 0-900; (od 0 do 900)
  • font-style: (styl czcionki)
    • normal; (normalna)
    • italic; (lekko pochylona)
    • oblique; (pochylona)
  • margin-strona: lub margin:(góra prawa dół lewa) (marginesy zewnętrzne)
    • left; (lewa)
    • right; (prawa)
    • top; (góra)
    • bottom; (dół)
  • padding-strona: lub padding:(góra prawa dół lewa) (marginesy wewnętrzne)
    • left; (lewa)
    • right; (prawa)
    • top; (góra)
    • bottom; (dół)
  • border: (grubość[px] rodzaj kolor);(ramka)
    • dotted; (kropkowana)
    • dashed; (przerywana)
    • solid; (ciągła)
    • double; (podwójna))
  • width: (szerokość elementu)
    • (ilość)px;
  • height: (wysokość elementu)
    • (ilość)px;
